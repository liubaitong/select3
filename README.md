select3
=======

带checkbox支持多选的select下拉框

使用方法
    $("#select").Select3({
        max:4,
        data: [
            {value: "1", text: "miao"},
            {value: "2", text: "zhuang"},
            {value: "3", text: "sailor"},
            {value: "4", text: "jack"},
            {value: "5", text: "roy"},
            {value: "6", text: "bill"}      ,
            {value: "7", text: "gates"},
            {value: "8", text: "jay"}

        ]
    });
